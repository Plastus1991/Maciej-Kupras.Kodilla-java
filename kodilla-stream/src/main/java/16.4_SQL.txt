SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS POSTS_NUMBER
FROM USERS U JOIN POSTS P ON U.ID = P.USER_ID
GROUP BY P.USER_ID
HAVING COUNT(*) > 1;
ORDER BY U.LASTNAME, U.FIRSTNAME;


SELECT I.SUMMARY, IL.NAME
FROM ISSUES I
JOIN ISSUESLISTS IL ON IL.ID = I.ISSUESLIST_ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_NUMBER
FROM ISSUES I JOIN USERS U ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
